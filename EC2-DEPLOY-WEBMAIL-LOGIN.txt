# Deploy custom webmail login to EC2

cd ~/mushqila

# Pull latest code
git pull

# Rebuild web container
docker-compose -f docker-compose.traefik.yml build web

# Restart
docker-compose -f docker-compose.traefik.yml up -d

# Wait
sleep 10

# Check logs
docker logs mushqila_web --tail=30

# Test webmail login at: https://mushqila.com/webmail/login/
